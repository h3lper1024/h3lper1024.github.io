    MySQL部署

    链接

    1.准备MySQL安装包

    2.检查是否已经部署过MySQL

    ps -ef | grep mysql

    rpm -qa | grep mysql

    3.上传MySQL安装包

    cd /usr/local

    rz

    4.解压

    tar -zxvf mysql-xxx.tar.gz

    5.创建软连接

    ln -s mysql-xxx mysql

    6.为MySQL创建用户组、用户

    groupadd -g 101 db

    useradd -u 514 -g db -G root -d /usr/local/mysql mysqladmin

    注：

    ​ groupadd

    ​ -g 指定用户组的id

    ​ useradd

    ​ -u 指定用户id

    ​ -g 指定用户主组

    ​ -G 指定用户附加组

    ​ -d 指定用户家目录

    如果用户mysqladmin已存在，执行：

    usermod -u 514 -g db -G root -d /usr/local/mysql mysqladmin

    7.恢复用户样式

    cp /etc/skel/.* /usr/local/mysql

    8.配置my.cnf文件

    vi /etc/my.cnf

    按i进入编辑模式

    复制：

    [client]
    port = 3306
    socket = /usr/local/mysql/data/mysql.sock

    [mysqld]
    port = 3306
    socket = /usr/local/mysql/data/mysql.sock

    skip-external-locking
    key_buffer_size = 256M
    sort_buffer_size = 2M
    read_buffer_size = 2M
    read_rnd_buffer_size = 4M
    query_cache_size= 32M
    max_allowed_packet = 16M
    myisam_sort_buffer_size=128M
    tmp_table_size=32M

    table_open_cache = 512
    thread_cache_size = 8
    wait_timeout = 86400
    interactive_timeout = 86400
    max_connections = 600

    #Try number of CPU’s*2 for thread_concurrency

    thread_concurrency = 32

    #isolation level and default engine
    default-storage-engine = INNODB
    transaction-isolation = READ-COMMITTED

    server-id = 1
    basedir = /usr/local/mysql
    datadir = /usr/local/mysql/data
    pid-file = /usr/local/mysql/data/hostname.pid

    #open performance schema
    log-warnings
    sysdate-is-now

    binlog_format = MIXED
    log_bin_trust_function_creators=1
    log-error = /usr/local/mysql/data/hostname.err
    log-bin=/usr/local/mysql/arch/mysql-bin
    #other logs
    #general_log =1
    #general_log_file = /usr/local/mysql/data/general_log.err
    #slow_query_log=1
    #slow_query_log_file=/usr/local/mysql/data/slow_log.err

    #for replication slave
    #log-slave-updates
    #sync_binlog = 1

    #for innodb options
    innodb_data_home_dir = /usr/local/mysql/data/
    innodb_data_file_path = ibdata1:500M:autoextend
    innodb_log_group_home_dir = /usr/local/mysql/arch
    innodb_log_files_in_group = 2
    innodb_log_file_size = 200M

    #TUN

    innodb_buffer_pool_size = 2048M
    innodb_additional_mem_pool_size = 50M
    innodb_log_buffer_size = 16M

    innodb_lock_wait_timeout = 100
    #innodb_thread_concurrency = 0
    innodb_flush_log_at_trx_commit = 1
    innodb_locks_unsafe_for_binlog=1

    #innodb io features: add for mysql5.5.8
    performance_schema
    innodb_read_io_threads=4
    innodb-write-io-threads=4
    innodb-io-capacity=200
    #purge threads change default(0) to 1 for purge
    innodb_purge_threads=1
    innodb_use_native_aio=on

    #case-sensitive file names and separate tablespace
    innodb_file_per_table = 1
    lower_case_table_names=1

    [mysqldump]
    quick
    max_allowed_packet = 16M

    [mysql]
    no-auto-rehash

    [mysqlhotcopy]
    interactive-timeout

    [myisamchk]
    key_buffer_size = 256M
    sort_buffer_size = 256M
    read_buffer = 2M
    write_buffer = 2M

    调整：

    根据实际情况，调整：

    innodb_buffer_pool_size = 2048M

    配置权限：

    chown mysqladmin:db /etc/my.cnf

    chmod 640 /etc/my.cnf

    ll my.cnf

    9.配置mysqladmin用户家目录权限

    chown -R mysqladmin:dba /usr/local/mysql
    chown -R mysqladmin:dba /usr/local/mysql/*
    chown -R mysqladmin:dba /usr/local/mysql-5.6.23-linux-glibc2.5-x86_64

    chmod -R 755 /usr/local/mysql
    chmod -R 755 /usr/local/mysql/*
    chmod -R 755 /usr/local/mysql-5.6.23-linux-glibc2.5-x86_64

    10.部署安装

    su - mysqladmin

    mkdir arch

    scripts/mysql_install_db

    注：这里可能会出现依赖问题

    scripts/mysql_install_db –user=mysqladmin –basedir=/usr/local/mysql –atadir=/usr/local/mysql/data

    11.配置开机自启动mysql

    cd /usr/local/mysql

    cp support-files/mysql.server /etc/rc.d/init.d/mysql

    chmod +x /etc/rc.d/init.d/mysql

    chkconfig –del mysql

    chkconfig –add mysql

    chkconfig –level 345 mysql on

    vi /etc/rc.local

    输入：

    su - mysqladmin -c “/etc/init.d/mysql start –federated”

    保存退出；

    12.启动MySQL

    su - mysqladmin

    rm -rf my.cnf my-new.cnf

    service mysql start

    13.配置数据库用户

    先用空密码登入mysql

    mysql -uroot -p

    回车登入

    use mysql;

    select host,user,password from user;

    delete from user where user=’’;

    update user set password=password(‘xxxxx’) where user=’root’;

    flush privileges;

    14.配置环境变量

    su - mysqladmin

    vi .bashrc

    输入：

    export MYSQL_HOME=/usr/local/mysql

    export PATH=$MYSQL_HOME/bin:$PATH

    保存退出

    source .bashrc

    echo $MYSQL_HOME

    which mysql

    15.优化mysqladmin用户样式

    su - mysqladmin

    vi .bashrc

    输入：

    PS1=uname -n“:”‘$USER’”:”‘$PWD’”:>”; export PS1

    保存退出

    source .bashrc

    16.重新部署
